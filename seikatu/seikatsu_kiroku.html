// ============================================================ // 生活記録アプリ - Google Apps Script バックエンド v4 // ============================================================ // 【設置手順】 // 1. Apps Script を開く（スプレッドシート > 拡張機能 > Apps Script） // 2. このコードを全選択して貼り付けて保存（Ctrl+S） // 3. デプロイ > デプロイを管理 > 鉛筆アイコン // > バージョン「新しいバージョン」> デプロイ // ============================================================ const EDITOR_PASSWORD = 'editor_pass_here'; // ← 変更してください const VIEWER_PASSWORD = 'viewer_pass_here'; // ← 変更してください const COLUMNS = [ 'date', 'breakfast', 'lunch', 'dinner', 'snack', 'weight', 'kot', 'outing', 'visitor', 'sleep', 'nap', 'dep1_time', 'dep1_dose', 'dep2_time', 'dep2_dose', 'kanpo1', 'kanpo2', 'kanpo3', 'condition', 'mood', 'fatigue', 'bath', 'hair', 'teeth', 'memo', 'trouble', 'updated_at' ]; // ------------------------------------------------------------ // 日付をYYYY-MM-DD文字列に統一（スプレッドシートの日付型対策） // ------------------------------------------------------------ function toDateString(val) { if (!val) return ''; if (typeof val === 'string' && /^\d{4}-\d{2}-\d{2}$/.test(val)) return val; try { var d = new Date(val); if (!isNaN(d.getTime())) { return Utilities.formatDate(d, 'Asia/Tokyo', 'yyyy-MM-dd'); } } catch(e) {} return String(val); } // ------------------------------------------------------------ // JSONレスポンス生成 // ------------------------------------------------------------ function makeResponse(obj) { var output = ContentService.createTextOutput(JSON.stringify(obj)); output.setMimeType(ContentService.MimeType.JSON); return output; } // ------------------------------------------------------------ // 認証 // ------------------------------------------------------------ function authenticate(role, password) { if (role === 'editor' && password === EDITOR_PASSWORD) return 'editor'; if (role === 'viewer' && password === VIEWER_PASSWORD) return 'viewer'; return null; } // ------------------------------------------------------------ // シート取得（年ごとに自動作成） // ------------------------------------------------------------ function getSheet(year) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheetName = String(year) + '年'; var sheet = ss.getSheetByName(sheetName); if (!sheet) { sheet = ss.insertSheet(sheetName); sheet.getRange(1, 1, 1, COLUMNS.length).setValues([COLUMNS]); sheet.setFrozenRows(1); sheet.getRange(1, 1, 1, COLUMNS.length) .setBackground('#5B8C7A') .setFontColor('white') .setFontWeight('bold'); sheet.getRange('A:A').setNumberFormat('@'); } return sheet; } // ============================================================ // GET：データ取得 // ============================================================ function doGet(e) { try { var p = e.parameter; var action = p.action || ''; var role = p.role || ''; var password = p.password || ''; var authed = authenticate(role, password); if (!authed) return makeResponse({ error: 'Unauthorized' }); if (action === 'getAll') return handleGetAll(); return makeResponse({ error: 'Unknown action: ' + action }); } catch(err) { return makeResponse({ error: err.toString() }); } } // ============================================================ // POST：データ保存 // application/x-www-form-urlencoded 形式で受信 // ============================================================ function doPost(e) { try { // URLSearchParams（フォームデータ）形式で受け取る var action = e.parameter.action || ''; var role = e.parameter.role || ''; var password = e.parameter.password || ''; var dataJson = e.parameter.data || '{}'; var authed = authenticate(role, password); if (!authed) return makeResponse({ error: 'Unauthorized' }); if (authed !== 'editor') return makeResponse({ error: 'Permission denied' }); if (action === 'save') { var data = JSON.parse(dataJson); return handleSave(data); } return makeResponse({ error: 'Unknown action: ' + action }); } catch(err) { return makeResponse({ error: err.toString() }); } } // ============================================================ // 全レコード取得 // ============================================================ function handleGetAll() { var ss = SpreadsheetApp.getActiveSpreadsheet(); var sheets = ss.getSheets(); var records = []; sheets.forEach(function(sheet) { if (!sheet.getName().match(/^\d{4}年$/)) return; var data = sheet.getDataRange().getValues(); if (data.length <= 1) return; var headers = data[0]; for (var i = 1; i < data.length; i++) { var rec = {}; headers.forEach(function(h, idx) { var val = data[i][idx]; rec[h] = (h === 'date') ? toDateString(val) : (val !== undefined ? String(val) : ''); }); if (rec.date) records.push(rec); } }); return makeResponse({ records: records }); } // ============================================================ // レコード保存（同一日付を完全に上書き） // ============================================================ function handleSave(data) { if (!data || !data.date) return makeResponse({ error: 'No date' }); var targetDate = data.date; var year = parseInt(targetDate.split('-')[0]); var sheet = getSheet(year); sheet.getRange('A:A').setNumberFormat('@'); data.updated_at = Utilities.formatDate( new Date(), 'Asia/Tokyo', 'yyyy/MM/dd HH:mm:ss' ); // 同一日付の行を下から全削除 var allData = sheet.getDataRange().getValues(); var deletedCount = 0; for (var i = allData.length - 1; i >= 1; i--) { if (toDateString(allData[i][0]) === targetDate) { sheet.deleteRow(i + 1); deletedCount++; } } // 末尾に追加 var rowValues = COLUMNS.map(function(col) { return data[col] !== undefined ? data[col] : ''; }); sheet.appendRow(rowValues); // 日付順ソート var lastRow = sheet.getLastRow(); if (lastRow > 2) { sheet.getRange(2, 1, lastRow - 1, COLUMNS.length).sort(1); } return makeResponse({ success: true, date: targetDate, deleted: deletedCount }); }
